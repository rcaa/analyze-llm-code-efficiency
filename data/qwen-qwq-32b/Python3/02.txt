from collections import defaultdict

def solve(words, puzzles):

    from collections import defaultdict

    freq = defaultdict(int)

    for word in words:

        mask = 0

        for c in word:

            mask |= 1 << (ord(c) - ord('a'))

        freq[mask] ] +=1

    res = []

    for p in puzzles:

        p_mask = 0

        for c in p:

            p_mask |= 1 << (ord(c) - ord('a')

        f_char = p[0]

        f_bit = 1 << (ord(f_char) - ord('a'))

        total =0

        s = p_mask

        while True:

            if (s & f_bit):

                total += freq.get(s, 0)

            if s ==0:

                break

            s = (s-1) & p_mask

        res.append(total)

    return res

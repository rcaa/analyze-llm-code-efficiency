Title: Number of Valid Words for Each Puzzle
URL: https://leetcode.com/problems/number-of-valid-words-for-each-puzzle/

from collections import defaultdict

def solve(words, puzzles):
    # Preprocess words into a dictionary mapping bitmasks to letter counts
    bitmask_counts = defaultdict(lambda: defaultdict(int))
    
    for word in words:
        mask = 0
        letters = set(word)
        for c in letters:
            mask |= 1 << (ord(c) - ord('a'))
        for c in letters:
            bitmask_counts[mask][c] += 1
    
    result = []
    for puzzle in puzzles:
        first_char = puzzle[0]
        puzzle_mask = 0
        for c in set(puzzle):
            puzzle_mask |= 1 << (ord(c) - ord('a'))
        
        total = 0
        subset = puzzle_mask
        while True:
            if subset in bitmask_counts:
                total += bitmask_counts[subset].get(first_char, 0)
            if subset == 0:
                break
            subset = (subset - 1) & puzzle_mask
        
        result.append(total)
    
    return result

explanation: The solution preprocesses each word into a bitmask and tracks how many times each letter appears for each bitmask. For each puzzle, it generates all subset bitmasks and efficiently counts valid words using precomputed data, ensuring high efficiency even for large inputs.

public class Solution {
    public int countPairs(String[] words) {
        Map<Integer, Integer> map = new HashMap<>();
        int result = 0;
        for (String s : words) {
            int n = s.length();
            // Compute prefix hashes and failure function
            long[] prefixHash = new long[n + 1];
            long[] power = new long[n + 1];
            int base = 911382629;
            int mod = 1000000000000000003;
            power[0] = 1;
            for (int i = 0; i < n; i++) {
                power[i + 1] = (power[i] * base) % mod;
            }
            for (int i = 0; i < n; i++) {
                prefixHash[i + 1] = (prefixHash[i] * base + s.charAt(i)) % mod;
            }
            // Compute failure function
            int[] failure = new int[n];
            int j = 0;
            for (int i = 1; i < n; i++) {
                while (j > 0 && s.charAt(i) != s.charAt(j)) {
                    j = failure[j - 1];
                }
                if (s.charAt(i) == s.charAt(j)) {
                    j++;
                }
                failure[i] = j;
            }
            // Collect all t lengths
            List<Integer> tLengths = new ArrayList<>();
            tLengths.add(n);
            int current = failure[n - 1];
            while (current > 0) {
                tLengths.add(current);
                current = failure[current - 1];
            }
            // For each t length, compute hash and look up in map
            for (int k : tLengths) {
                if (k == 0) continue;
                long hash = getHash(prefixHash, power, 0, k - 1);
                result += map.getOrDefault((int) hash, 0);
            }
            // Add the current string's hash to the map
            long hashS = getHash(prefixHash, power, 0, n - 1);
            map.put((int) hashS, map.getOrDefault((int) hashS, 0) + 1);
        }

public class Solution {
    public int countPairs(String[] words) {
        Map<String, Integer> map = new HashMap<>();
        int result = 0;
        for (String s : words) {
            int n = s.length();
            // Compute failure function
            int[] failure = computeFailureFunction(s);
            // Collect all t's
            Set<String> tSet = new HashSet<>();
            // Add the entire string
            tSet.add(s);
            int current = failure[n - 1];
            while (current > 0) {
                tSet.add(s.substring(0, current));
                current = failure[current - 1];
            }
            // For each t in tSet, add the count from map
            for (String t : tSet) {
                result += map.getOrDefault(t, 0);
            }
            // Add s to the map
            map.put(s, map.getOrDefault(s, 0) + 1);
        }
        return result;
    }

    private int[] computeFailureFunction(String s) {
        int n = s.length();
        int[] failure = new int[n];
        int j = 0;
        for (int i = 1; i < n; i++) {
            while (j > 0 && s.charAt(i) != s.charAt(j)) {
                j = failure[j - 1];
            }
            if (s.charAt(i) == s.charAt(j)) {
                j++;
            }
            failure[i] = j;
        }
        return failure;
    }

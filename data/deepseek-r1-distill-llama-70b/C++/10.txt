Title: Maximize Count of Distinct Primes After Split
URL: https://leetcode.com/problems/maximize-count-of-distinct-primes-after-split/
Status: OK

solution: 
```cpp
#include <vector>
#include <unordered_set>
#include <algorithm>

using namespace std;

vector<bool> sieve;

void compute_sieve(int max_val) {
    sieve.resize(max_val + 1, true);
    sieve[0] = sieve[1] = false;
    for (int i = 2; i * i <= max_val; ++i) {
        if (sieve[i]) {
            for (int j = i * i; j <= max_val; j += i) {
                sieve[j] = false;
            }
        }
    }
}

int main() {
    // Read input
    int n;
    vector<int> nums(n);
    int q;
    vector<vector<int>> queries(q, vector<int>(2));

    compute_sieve(1e5);

    for (auto& query : queries) {
        int idx = query[0];
        int val = query[1];
        nums[idx] = val;

        vector<int> prefix_primes;
        unordered_set<int> seen;
        for (int i = 0; i < n; ++i) {
            if (sieve[nums[i]]) {
                seen.insert(nums[i]);
            }
            prefix_primes.push_back(seen.size());
        }

        vector<int> suffix_primes;
        unordered_set<int> seen_suffix;
        for (int i = n - 1; i >= 0; --i) {
            if (sieve[nums[i]]) {
                seen_suffix.insert(nums[i]);
            }
            suffix_primes.push_back(seen_suffix.size());
        }
        reverse(suffix_primes.begin(), suffix_primes.end());

        int max_sum = 0;
        for (int k = 1; k < n; ++k) {
            int current = prefix_primes[k - 1] + suffix_primes[k];
            if (current > max_sum) {
                max_sum = current;
            }
        }

        // Handle k = n case (though k < n)
        // So, the loop covers all possible k

        cout << max_sum << endl;
    }

    return 0;
}
```

efficiency: LOW

time complexity: O(n * q)

space complexity: O(n)

energy implications: HIGH

explanation: This solution updates the array for each query and then recomputes the prefix and suffix counts of distinct primes. It iterates through all possible split points to find the maximum sum, resulting in O(n) time per query, which is inefficient for large inputs but straightforward to implement.

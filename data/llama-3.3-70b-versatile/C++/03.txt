Title: Construct Target Array With Multiple Sums
URL: https://leetcode.com/problems/construct-target-array-with-multiple-sums/
Status: OK

solution:
class Solution {
public:
    bool isPossible(vector<int>& target) {
        priority_queue<long long> pq;
        long long sum = 0;
        for (int i = 0; i < target.size(); i++) {
            sum += target[i];
            pq.push(target[i]);
        }
        while (pq.top() != 1) {
            long long x = pq.top();
            pq.pop();
            sum -= x;
            if (x <= sum || sum < 1) return false;
            x %= sum;
            if (x == 0) x = sum;
            sum += x;
            pq.push(x);
        }
        return true;
    }
};

efficiency: HIGH
time complexity: O(n log n)
space complexity: O(n)
energy implications: LOW
explanation: The solution uses a priority queue to efficiently manage the array elements, repeatedly replacing the largest element with its remainder when divided by the sum of the other elements, until all elements are 1 or it becomes impossible to reach the target array.

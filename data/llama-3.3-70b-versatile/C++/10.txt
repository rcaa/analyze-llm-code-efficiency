Title: Maximize Count of Distinct Primes After Split
URL: https://leetcode.com/problems/maximize-count-of-distinct-primes-after-split/
Status: OK

solution:
```#include <iostream>
#include <vector>
#include <unordered_set>
#include <cmath>

bool isPrime(int n) {
    if (n <= 1) return false;
    if (n == 2) return true;
    if (n % 2 == 0) return false;
    for (int i = 3; i * i <= n; i += 2) {
        if (n % i == 0) return false;
    }
    return true;
}

std::vector<int> maxSumOfDistinctPrimes(std::vector<int>& nums, std::vector<std::vector<int>>& queries) {
    std::vector<int> result;
    for (auto& query : queries) {
        nums[query[0]] = query[1];
        int maxCount = 0;
        for (int k = 1; k < nums.size(); k++) {
            std::unordered_set<int> prefix, suffix;
            for (int i = 0; i < k; i++) {
                if (isPrime(nums[i])) prefix.insert(nums[i]);
            }
            for (int i = k; i < nums.size(); i++) {
                if (isPrime(nums[i])) suffix.insert(nums[i]);
            }
            maxCount = std::max(maxCount, (int)prefix.size() + (int)suffix.size());
        }
        result.push_back(maxCount);
    }
    return result;
}

efficiency: MEDIUM
time complexity: O(q * n * sqrt(m))
space complexity: O(n)
energy implications: MEDIUM
explanation: The solution iterates over each query, updates the array, and then checks all possible splits of the array to find the maximum sum of distinct primes in each part. The isPrime function checks if a number is prime, and the maxSumOfDistinctPrimes function returns the result for each query.

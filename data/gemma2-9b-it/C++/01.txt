Title: Online Majority Element In Subarray
URL: https://leetcode.com/problems/online-majority-element-in-subarray/
Status: OK

solution: 
```cpp
class MajorityChecker {
public:
    vector<int> count;
    MajorityChecker(vector<int>& arr) {
        count.resize(100005, 0);
        for (int num : arr) {
            count[num]++;
        }
    }
    
    int query(int left, int right, int threshold) {
        int maxCount = 0;
        int majority = -1;
        for (int i = left; i <= right; i++) {
            int num = arr[i];
            if (count[num] > maxCount) {
                maxCount = count[num];
                majority = num;
            }
        }
        return (maxCount >= threshold) ? majority : -1;
    }
private:
    vector<int> arr;
};
```
efficiency: HIGH
time complexity: O(N)
space complexity: O(N)
energy implications: LOW
explanation: The solution utilizes a frequency map to store the count of each element in the input array. The `query` function iterates through the subarray, updating the maximum count and corresponding majority element. The efficiency stems from the pre-computed frequency map, enabling constant-time lookups for element counts.
